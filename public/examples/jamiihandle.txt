

ID=
valg=

if [ "$1" = "-ID" ]; then
	shift
	ID="$1"
	shift
else
	echo "No ID??"
	read
	exit
fi

until [ "$valg" = "0" ]; do
	clear
	echo "Tast 1 for at sælge"
	echo "Tast 2 for at Købe"
	echo "Tast 0 for at logge ud"
	read valg
	case $valg in
		1)	bash "/usr/local/bin/troels" -ID $ID
			;;
		2)	bash "/usr/local/bin/amalie" -ID $ID
			;;
		0)	
			;;
		*)	echo "Du skal trykke 1, 2 eller 0 - Tryk enter for at prøve igen"
			read -s
	esac
done
